Hrho=9000.  # m

def calc_tau(k,rmix,rho_air,Hrho,z):
    """
    all units mks
    """
    tau_z = k*rmix*rho_air*Hrho*(1 - np.exp(-(z/Hrho)))
    return tau_z

def calc_trans(k,rmix,rho_air,Hrho,z,ztop):
    tau_tot = calc_tau(k,rmix,rho_air,Hrho,ztop)
    tau_z = calc_tau(k,rmix,rho_air,Hrho,z)
    trans_z = np.exp(-(tau_tot - tau_z))
    return trans_z

#
# transmissivity from 1 - 6 km

trans_z_layer = calc_trans(k,rmix,rho_air,Hrho,z_layer_m,ztop)

def find_slope(t,B):
    #
    # find the delta values last - first
    #
    delta_t = t[-1] - t[0]
    delta_B = B[-1] - B[0]
    slope = delta_B/delta_t
    return slope

meter2micron = 1.e-6

Q1b – analytic solution#

b) Given (55) derive an analytic solution for the radiance \(L\) at 6 km coming from the air between 1 km and 6 km. That is, solve the Schwartzchild integral:

(56)#\[ L(6km) = \int_{1km}^{6km} B(t^\prime) dt^\prime \]

as an equation.

Q1b solution – equations#

Integrate (56) and get:

(57)#\[\begin{split} \begin{align} &\int_{1km}^{6km} \left ( B_0+\Gamma_t t^\prime \right )dt^\prime = \\ &B_0(t_{6km} - t_{1km}) + \left ( \frac{t_{6km}^2}{2} - \frac{t_{1km}^2}{2} \right ) \Gamma_t \end{align} \end{split}\]

where \(B_0\) is given by:

(58)#\[\begin{split} \begin{aligned} & B=B_{3.5 \mathrm{~km}}+\left(t-t_{3.5}\right) \Gamma_t \\ & B=\left(B_{3.5 \mathrm{~km}}+\Gamma_t\left(-t_{3.5}\right)\right)+\Gamma_t t \\ & \text{therefore} \\ & B_0=B_{3.5 \mathrm{~km}}-t_{3.5} \Gamma_t \end{aligned} \end{split}\]

In another cell in your notebook, calculate L(6km) for the three values of \(\Gamma_z\) using your solution to (56). How much do they vary, in percent, from each other?

def calc_B0(B35,trans35,gamma_t):
    B0 = B35 - trans35*gamma_t
    return B0
    
def calc_L6km(B35,trans35, trans_z_layer ,gamma_t):
    """
    all units are mks
    """
    trans6km = trans_z_layer[-1]
    trans1km = trans_z_layer[0]
    delta_trans = trans6km - trans1km
    B0 = calc_B0(B35,trans35,gamma_t)
    term1 = B0*delta_trans
    term2 = (trans6km**2.- trans1km**2.)/2.*gamma_t
    answer = term1 + term2
    return answer

def find_ndvi(out_dict):
    band4 = out_dict['b4_ds']
    band5 = out_dict['b5_ds']
    fmask = out_dict['fmask_ds']
    ndvi = (band5 - band4)/(band5 + band4)
    ndvi = ndvi*fmask
    return ndvi
